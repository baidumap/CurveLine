 20     /**

 24         /**
 25          * CurveLine类的构造函数
 26          * @class 弧线类，实现效果的<b>入口</b>。
 27          * 实例化该类后，即可返回一个弧线的Polyline对象，使用方法同<a href="http://developer.baidu.com/map/reference/index.php?title=Class:%E8%A6%86%E7%9B%96%E7%89%A9%E7%B1%BB/Polyline" target="_blank">Polyline</a>
 28          * 即可调用map.addOverlay方法添加到地图当中
 29          * 
 30          * @constructor
 31          * @param {Array<Point>} Array<Point> Point数组对象
 32          * @param {Json Object} opts 可选的输入参数，非必填项。可输入选项参考<a href="http://developer.baidu.com/map/reference/index.php?title=Class:%E8%A6%86%E7%9B%96%E7%89%A9%E7%B1%BB/PolylineOptions" target="_blank">PolylineOptions</a><br />
 33          *
 34          * @example <b>参考示例：</b><br />
 35          * var map = new BMap.Map("container");<br />map.centerAndZoom(new BMap.Point(116.404, 39.915), 15);<br/>var points = [new BMap.Point(116.432045,39.910683), new BMap.Point(116.388522,39.985964), <br/>new BMap.Point(117.218862,39.141468), new BMap.Point(121.485947,31.510083)];
 36          * <br />var curve = new BMapLib.CurveLine(points, {strokeColor:"blue", strokeWeight:3, strokeOpacity:0.5}); //新建弧线覆盖物对象
 37          * <br />map.addOverlay(curve); //添加到地图
 38          * <br/>curve.enableEditing(); //开启编辑功能
 39          */
 40         BMapLib.CurveLine = function(points, opts){
 41             var curvePoints = getCurvePoints(points);
 42             var polyline = new BMap.Polyline(curvePoints, opts);
 43             polyline.addEventListener('lineupdate', function(){
 44                 if (this.isEditing) {
 45                     this.enableEditing();
 46                 }
 47             });
 48             polyline.editMarkers = [];
 49             polyline.points = points;
 50             polyline.enableEditing = enableEditing;
 51             polyline.disableEditing = disableEditing;
 52             polyline.getPath = getPath;
 53             return polyline;
 54         }

 56         /**
 57          * 根据弧线的坐标节点数组
 58          */
 59         getCurvePoints = function(points) {
 60             var curvePoints = [];
 61             for (var i = 0; i < points.length - 1; i++) {
 62                 var p = getCurveByTwoPoints(points[i], points[i + 1]);
 63                 if (p && p.length > 0) {
 64                     curvePoints = curvePoints.concat(p);
 65                 }
 66             }
 67             return curvePoints;
 68         }
 69 
 70         /**
 71          * 根据两点获取曲线坐标点数组
 72          * @param Point 起点
 73          * @param Point 终点
 74          */
 75         getCurveByTwoPoints = function(obj1, obj2) {
 76             if (!obj1 || !obj2 || !(obj1 instanceof BMap.Point) || !(obj2 instanceof BMap.Point)) {
 77                 return null;
 78             }
 79 
 80             var B1 = function(x) {
 81                 return 1 - 2 * x + x * x;
 82             };
 83             var B2 = function(x) {
 84                 return 2 * x - 2 * x * x;
 85             };
 86             var B3 = function(x) {
 87                 return x * x;
 88             };
 89 
 90             curveCoordinates = [];
 91 
 92             var count=30; // 曲线是由一些小的线段组成的，这个表示这个曲线所有到的折线的个数
 93             var isFuture=false;
 94             var t, h, h2, lat3, lng3, j, t2;
 95             var LnArray = [];
 96             var i = 0;
 97             var inc = 0;
 98 
 99             if (typeof(obj2) == "undefined") {
100                 if (typeof(curveCoordinates) != "undefined") {
101                     curveCoordinates = [];
102                 }
103                 return;
104             }
105 
106             var lat1 = parseFloat(obj1.lat);
107             var lat2 = parseFloat(obj2.lat);
108             var lng1 = parseFloat(obj1.lng);
109             var lng2 = parseFloat(obj2.lng);
110                 
111             // 计算曲线角度的方法
112             if (lng2 > lng1) {
113                 if (parseFloat(lng2-lng1) > 180) {
114                     if (lng1 < 0) {
115                         lng1 = parseFloat(180 + 180 + lng1);
116                     }
117                 }
118             }
119             
120             if (lng1 > lng2) {
121                 if (parseFloat(lng1-lng2) > 180) {
122                     if (lng2 < 0) {
123                         lng2 = parseFloat(180 + 180 + lng2);
124                     }
125                 }
126             }
127             j = 0;
128             t2 = 0;
129             if (lat2 == lat1) {
130                 t = 0;
131                 h = lng1 - lng2;
132             } else if (lng2 == lng1) {
133                 t = Math.PI / 2;
134                 h = lat1 - lat2;
135             } else {
136                 t = Math.atan((lat2 - lat1) / (lng2 - lng1));
137                 h = (lat2 - lat1) / Math.sin(t);
138             }
139             if (t2 == 0) {
140                 t2 = (t + (Math.PI / 5));
141             }
142             h2 = h / 2;
143             lng3 = h2 * Math.cos(t2) + lng1;
144             lat3 = h2 * Math.sin(t2) + lat1;
145 
146             for (i = 0; i < count + 1; i++) {
147                 curveCoordinates.push(new BMap.Point(
148                     (lng1 * B1(inc) + lng3 * B2(inc)) + lng2 * B3(inc),
149                     (lat1 * B1(inc) + lat3 * B2(inc) + lat2 * B3(inc))
150                 ));
151                 inc = inc + (1 / count);
152             }
153             return curveCoordinates;
154         }

156         /**
157          * 重写弧线的编辑功能
158          */
159         function enableEditing() {
160             if (this.map) {
161                 this.disableEditing();
162                 for (var i = 0; i < this.points.length; i++) {
163                     var marker = new BMap.Marker(this.points[i], {
164                         icon: new BMap.Icon('../src/circle.png', new BMap.Size(16,16)),
165                         enableDragging: true,
166                         raiseOnDrag: true
167                     });
168                     var self = this;
169                     marker.addEventListener('dragend', function(){
170                         self.points.length = 0;
171                         for (var i = 0; i < self.editMarkers.length; i++) {
172                             self.points.push(self.editMarkers[i].getPosition());
173                         }
174                         var curvePoints = getCurvePoints(self.points)
175                         self.setPath(curvePoints);
176                     });
177                     marker.index = i;
178                     this.editMarkers.push(marker);
179                     this.map.addOverlay(marker);
180                 }
181             }
182             this.isEditing = true;
183         }
184 
185         /**
186          * 重写弧线的编辑功能
187          */
188         function disableEditing() {
189             this.isEditing = false;
190             //清空之前的编辑点
191             for (var i = 0; i < this.editMarkers.length; i++) {
192                 this.map.removeOverlay(this.editMarkers[i]);
193                 this.editMarkers[i] = null;
194             }
195             this.editMarkers.length = 0;
196         }
197 
198         /**
199          * 获取弧线的坐标点
200          */
201         function getPath() {
202             return this.points;
203         }
